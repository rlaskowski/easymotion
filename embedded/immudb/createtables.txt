CREATE TABLE IF NOT EXISTS user (
    id INTEGER AUTO_INCREMENT,
    name VARCHAR[30],
    email VARCHAR[60],
    password VARCHAR[30],
    created TIMESTAMP,
    PRIMARY KEY id
);

CREATE TABLE IF NOT EXISTS auth_groups (
    id INTEGER AUTO_INCREMENT,
    name VARCHAR[30],
    created TIMESTAMP,
    PRIMARY KEY id
);

CREATE TABLE IF NOT EXISTS user_groups (
    user_id INTEGER,
    group_id INTEGER,
    created TIMESTAMP,
    PRIMARY KEY (user_id, group_id)
);

CREATE TABLE IF NOT EXISTS camera_options (
    id INTEGER AUTO_INCREMENT,
    name VARCHAR[30],
    camera_id INTEGER,
    auto_recording BOOLEAN,
    timeline BOOLEAN,
    PRIMARY KEY id
)
